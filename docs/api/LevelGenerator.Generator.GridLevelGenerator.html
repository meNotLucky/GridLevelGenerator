    Examples

    This sample shows how to call the GenerateNewLevel() method from an ExampleClass component on a GameObject that also has a GridLevelGenerator component.

    class ExampleClass : MonoBehaviour
{
    private GridLevelGenerator _generator;

    private void Start()
    {
        _generator = GetComponent<GridLevelGenerator>();
        _generator.GenerateNewLevel;
    }
}
